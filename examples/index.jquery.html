jQuery Plugin used with regular jquery

This example showcase the plugin used with "normal" jquery.

With data-attributes

Usage

<div data-norwegian-libraries='{"query":"title or ISBN", "labelText": "Finn boka", "labelClass": "", "selectClass": "form-control", "btnString":"Finn boka", "btnClass": "btn btn-primary", "favString": "Sett som favoritt", "favClass": "text-muted", "formClass":"form-inline", "cookieName" : "finn-boka-fav"}'></div>

With regular jQuery

Event outbut

Change select box or submit to see output

Usage

$('.js-event').norwegianLibraries({
            query:"Tittel eller ISBN",
            labelText: "Finn boka",
            labelClass: "label-class",
            selectClass: "form-control",
            btnString:"Finn boka",
            btnClass: "btn btn-primary",
            favString: "Sett som favoritt",
            favClass: "text-muted",
            formClass:"form-inline",
            cookieName : "finn-boka-fav",
            onSubmit : function(library) {
                console.log(library)
            },
            onSelect : function(library) {
                console.log(library)
            }
        });
