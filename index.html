Norwegian Libraries jQuery Plugin

jQuery Plugin to find books in norwegian libraries.

Install and usage

npm i norwegian-libraries
var $ = require('jquery');
require("norwegian-libraries");

$(function() {
	$('.js-event').each(function(){
		var me = $(this),
			query = me.text();
		me.norwegianLibraries({
			query:"title or ISBN"
		});
	});
});
